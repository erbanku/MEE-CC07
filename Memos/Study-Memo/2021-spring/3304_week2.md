import RPi.GPIO as GPIO
import time
GPIO.setmode(GPIO.BOARD)
GPIO.setup(22,GPIO.OUT)
GPIO.setup(18,GPIO.OUT)
GPIO.setup(16,GPIO.IN)
num = 0
m=0
n=0
while True:
    if GPIO.input(16)==1:
        num = (num+1) % 4
        print(num)
        time.sleep(0.5)
    m=num/2
    n=num%2
    if m==0:
        GPIO.output(22,0)
    else:
        GPIO.output(22,1)
    if n==0:
        GPIO.output(18,0)
    else:
        GPIO.output(18,1)
